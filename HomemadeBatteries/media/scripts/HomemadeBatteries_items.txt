module HomemadeBatteries
{
	/*
		Tier 1: cobbled together from scrap metal, wire and acid.
		Drains noticeably faster than a factory battery (UseDelta 0.02 vs vanilla 0.007)
		to reflect it being a crude, homemade job.
	*/
	item HomemadeBattery
	{
		DisplayCategory = Electronics,
		ItemType = base:drainable,
		DisplayName = Homemade Battery,
		Weight = 0.3,
		Icon = Homemade,
		MetalValue = 5.0,
		UseDelta = 0.02,
		UseWhileEquipped = false,
		cantBeConsolided = true,
		WorldStaticModel = Battery,
		Tags = base:hasmetal,
	}

	/*
		Tier 2: the Homemade Battery reworked with extra components and a
		steadier hand. Holds charge much closer to a real battery.
	*/
	item ImprovedBattery
	{
		DisplayCategory = Electronics,
		ItemType = base:drainable,
		DisplayName = Improved Homemade Battery,
		Weight = 0.5,
		Icon = Improved,
		MetalValue = 8.0,
		UseDelta = 0.009,
		UseWhileEquipped = false,
		cantBeConsolided = true,
		WorldStaticModel = Battery,
		Tags = base:hasmetal,
	}
}
